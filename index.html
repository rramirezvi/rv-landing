CREATE OR REPLACE FUNCTION eficiencia_mes(mes int, anio int)
RETURNS TABLE(gestor text, meta numeric, recupero numeric, eficiencia text) AS $$
BEGIN
  RETURN QUERY
  SELECT g.nombre, m.meta_1, SUM(p.monto),
         ROUND(SUM(p.monto)*100/m.meta_1, 2) || '%'
  FROM metas m
  JOIN pagos p ON p.fecha BETWEEN m.fecha_inicio AND m.fecha_cierre
  JOIN gestores g ON g.id = p.id_gestor
  GROUP BY g.nombre, m.meta_1;
END; $$ LANGUAGE plpgsql;
